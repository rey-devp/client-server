// State Machine Algorithm
// Client-Server Implementation
 
let currentState = IDLE
 
function handleEvent(event) {
  switch (currentState) {
 
    case IDLE:
      if (event == 'USER_SEND') {
        currentState = REQUEST
        buildHttpRequest()
      } break
 
    case REQUEST:
      // paket melintasi jaringan TCP/IP
      if (event == 'PACKET_ARRIVED') {
        currentState = PROCESS
        runBusinessLogic()
      } break
 
    case PROCESS:
      // query database di server
      if (event == 'DB_QUERY_DONE') {
        currentState = RESPONSE
        buildHttpResponse()
      } break
 
    case RESPONSE:
      // respons HTTP dikirim ke klien
      if (event == 'RESPONSE_RECEIVED') {
        currentState = IDLE
        updateClientUI()
      } break
 
  } // end switch
}
