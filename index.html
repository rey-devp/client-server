Pseudocode

// State Machine Algorithm
// Client-Server Implementation
 
let currentState = IDLE
 
function handleEvent(e) {
  switch (currentState) {
 
    case IDLE:
      if (e == 'USER_SEND') {
        currentState = REQUEST
        buildRequest()
      } break
 
    case REQUEST:
      // paket → jaringan
      if (e == 'PKT_ARRIVED') {
        currentState = PROCESS
        runLogic()
      } break
 
    case PROCESS:
      // server + DB query
      if (e == 'DB_DONE') {
        currentState = RESPONSE
        buildResponse()
      } break
 
    case RESPONSE:
      // respons → klien
      if (e == 'RESP_RECEIVED') {
        currentState = IDLE
        updateUI()
      } break
 
  } // end switch
}
